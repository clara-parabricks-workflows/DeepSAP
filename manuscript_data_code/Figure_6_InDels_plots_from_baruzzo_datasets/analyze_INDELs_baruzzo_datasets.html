def collect_benchmarking_stats(base_dir, aligner_names, sample_ids):
    """
    Collect benchmarking statistics for multiple aligners and samples.

    Args:
        base_dir (str): The base directory containing benchmarking files.
        aligner_names (list): List of aligner names.
        sample_ids (list): List of sample IDs.

    Returns:
        pd.DataFrame: A DataFrame with collected stats, including:
                      ['sample_id', 'aligner_name', 'metric', 'value'].
    """
    # Initialize a list to store the results
    stats = []

    # Regular expressions to extract the desired metrics
    metrics_regex = {
        "insertions TP": re.compile(r"insertions TP:\s+(\d+)"),
        "insertions FN": re.compile(r"insertions FN:\s+(\d+)"),
        "insertions FP": re.compile(r"insertions FP:\s+(\d+)"),
        "deletions TP": re.compile(r"deletions TP:\s+(\d+)"),
        "deletions FN": re.compile(r"deletions FN:\s+(\d+)"),
        "deletions FP": re.compile(r"deletions FP:\s+(\d+)"),
    }

    # Iterate through each sample and aligner
    for sample_id in sample_ids:
        for aligner_name in aligner_names:
            # Construct the path to the benchmarking file
            file_path = os.path.join(
                base_dir, sample_id, f"{sample_id}_{aligner_name}_benchmark.txt"
            )
            # print(file_path)
            if os.path.exists(file_path):
                # Read the file content
                with open(file_path, "r") as f:
                    content = f.read()
                # print(content)
                # Extract metrics from the file content
                for metric, regex in metrics_regex.items():
                    match = regex.search(content)
                    if match:
                        value = int(match.group(1))  # Convert the value to integer
                        # Add the metric to the stats
                        stats.append({
                            "sample_id": sample_id,
                            "aligner_name": aligner_name,
                            "metric": metric,
                            "value": value,
                        })
            else:
                print(f"Warning: Benchmarking file not found: {file_path}")

    # Convert stats to a DataFrame
    if stats:
        stats_df = pd.DataFrame(stats)
    else:
        raise FileNotFoundError("No benchmarking files found in the provided directory.")

    # Return the final DataFrame
    return stats_df

def compare_aligners_to_truth(cig_file, aligner_list, ruby_fix_script_path, ruby_compare_to_truth_script, output_path, sample_id):
    """
    Compare each aligner's SAM output to the ground truth CIG file.
    
    Args:
        cig_file (str): Path to the ground truth .cig file.
        aligner_list (list of lists): Each sublist contains [aligner_name, bam_path].
        ruby_fix_script_path (str): Path to the Ruby script for fixing the SAM file.
        ruby_compare_to_truth_script (str): Path to the Ruby script for comparing with ground truth.
        output_path (str): Directory where output files will be saved.
        
    """
    # Ensure the output directory exists
    os.makedirs(output_path, exist_ok=True)
    
    # Use ThreadPoolExecutor to run each aligner comparison in parallel
    with ThreadPoolExecutor() as executor:
        futures = [
            executor.submit(process_aligner, cig_file, bam_path, sample_id, aligner_name, ruby_fix_script_path, ruby_compare_to_truth_script, output_path)
            for aligner_name, bam_path in aligner_list
        ]
        
        # Optionally, wait for all threads to complete and handle any exceptions
        for future in futures:
            try:
                future.result()  # Retrieve the result to capture exceptions, if any
            except Exception as e:
                print(f"An error occurred: {e}")

def plot_indel_metrics_grouped(df):
    """
    Generate a 2x2 grid of grouped stacked bar charts for INDEL metrics (TP, FP, FN)
    for "human" and "malaria" samples, grouped by aligner with abbreviated dataset names on the x-axis.
    The stacked bars are shown as percentages, with the total of TP, FP, and FN summing to 100% for each sample-aligner.

    Parameters:
        df (pd.DataFrame): A dataframe with columns ['sample_id', 'aligner_name', 'metric', 'value'].
    """
    plt.rcParams["font.family"] = "DejaVu Sans"
    metrics = ["TP", "FP", "FN"]
    variants = ["insertions", "deletions"]
    groups = ["human", "malaria"]
    colors = {"TP": "#019E72", "FP": "#C0C0C0", "FN": "#CD3701"}  # New color mapping for each metric

    # Mapping for dataset abbreviations
    dataset_abbreviation = {
        "baruzzo_human_t3r1": "T3",
        "baruzzo_human_t2r1": "T2",
        "baruzzo_human_t1r1": "T1",
        "baruzzo_malaria_t3r1": "T3",
        "baruzzo_malaria_t2r1": "T2",
        "baruzzo_malaria_t1r1": "T1"
    }

    # Create a 2x2 grid of plots with additional spacing
    fig, axes = plt.subplots(2, 2, figsize=(20, 13))  # Increase height for better spacing
    axes = axes.flatten()

    for idx, (variant, group) in enumerate([(v, g) for g in groups for v in variants]):
        ax = axes[idx]

        # Filter data for the current group and variant
        group_df = df[df["sample_id"].str.contains(group) & df["metric"].str.contains(variant)]
        
        # Get unique aligners and samples
        aligners = group_df["aligner_name"].unique()
        samples = group_df["sample_id"].unique()

        # Prepare x-axis positions for grouped bar chart
        aligner_positions = np.arange(len(aligners)) * (len(samples) + 2)  # Space between aligner groups
        x = []
        for aligner_pos in aligner_positions:
            x.extend(aligner_pos + np.arange(len(samples)))

        bar_width = 0.8
        spacing_within_cluster = 0.3

        # Build the stacked bars
        for i, sample in enumerate(samples):
            bottom_values = np.zeros(len(aligners))
            for metric in metrics:
                metric_values = [
                    group_df[(group_df["sample_id"] == sample) & 
                             (group_df["aligner_name"] == aligner) & 
                             (group_df["metric"] == f"{variant} {metric}")
                            ]["value"].values[0]
                    if not group_df[(group_df["sample_id"] == sample) & 
                                    (group_df["aligner_name"] == aligner) & 
                                    (group_df["metric"] == f"{variant} {metric}")
                                   ].empty
                    else 0
                    for aligner in aligners
                ]

                # Calculate the total (TP + FP + FN) for each sample-aligner pair
                total_values = np.array([sum([group_df[(group_df["sample_id"] == sample) & 
                                                        (group_df["aligner_name"] == aligner) & 
                                                        (group_df["metric"] == f"{variant} {m}")]["value"].values[0] 
                                                        if not group_df[(group_df["sample_id"] == sample) & 
                                                                        (group_df["aligner_name"] == aligner) & 
                                                                        (group_df["metric"] == f"{variant} {m}")]
                                                                   .empty else 0 for m in metrics])
                                        for aligner in aligners])

                # Convert counts to percentages for each metric
                metric_values = [
                    metric_value / total_value * 100 if total_value > 0 else 0
                    for metric_value, total_value in zip(metric_values, total_values)
                ]

                bar_positions = x[i::len(samples)]
                ax.bar(
                    bar_positions,
                    metric_values,
                    bar_width,
                    bottom=bottom_values,
                    label=metric if i == 0 else None,
                    color=colors[metric]  # Use new color mapping
                )
                bottom_values += metric_values

        # Add aligner group labels
        for aligner_pos, aligner in zip(aligner_positions, aligners):
            ax.text(
                aligner_pos + (len(samples) - 1) / 2,
                ax.get_ylim()[1] * 0.955,
                aligner,
                ha="center",
                va="bottom",
                fontsize=10,
                weight="bold"
            )

        # Set x-axis ticks and labels (dataset abbreviations)
        ax.set_xticks(x)
        ax.set_xticklabels(
            [dataset_abbreviation.get(sample, sample) for sample in samples] * len(aligners),
            rotation=0,
            fontsize=10
        )
        print("variant", variant)
        if variant == "insertions":
            ax.set_ylabel("Percentage of inserted bases", fontsize=12)
        else:
            ax.set_ylabel("Percentage of deleted bases", fontsize=12)

        # ax.set_ylabel("Percentage (%)", fontsize=12)
        ax.set_title(f"{group.capitalize()} {variant.capitalize()}", fontsize=14, weight="bold")
        ax.grid(axis="y", linestyle="--", alpha=0.7)

    # Add a shared legend
    handles, labels = axes[0].get_legend_handles_labels()
    fig.legend(handles, labels, title="Metrics", loc="upper center", bbox_to_anchor=(0.5, 0.97), ncol=len(metrics))

    # Adjust layout to add space between plots
    plt.tight_layout(rect=[0, 0, 1, 0.93], h_pad=2.5, w_pad=2.5)  # Add padding between rows and columns
    plt.show()
    output_file = "INDELS_per_alinger_percentage.png"
    fig.savefig(output_file, dpi=300, bbox_inches='tight')
